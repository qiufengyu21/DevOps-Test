exports.openStudy = function(req, res) {
    var token = req.body.token;
    db.collection('studies', function(err, collection) {
        collection.findOne({'token':token}, function(err, study) {
            collection.update( {'_id' : study._id}, 
                     {'$set' : {'status' : 'open'}});
            res.send({status:'ok'});
        });
    });
}
